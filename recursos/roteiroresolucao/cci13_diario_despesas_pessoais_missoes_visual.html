3void registrarDespesa() {
4 char data[11], categoria[30];
5 double valor;
6 printf("Data (DD/MM/AAAA): "); scanf("%10s", data);
7 printf("Categoria (uma palavra): "); scanf("%29s", categoria);
8 printf("Valor: R$ "); scanf("%lf", &valor);
9 FILE *arquivo = fopen("despesas.txt", "a");
10 if (arquivo == NULL) {
11 printf("Nao foi possivel abrir o arquivo.\n");
12 return;
13 }
14 fprintf(arquivo, "%s %s %.2f\n", data, categoria, valor);
15 if (fclose(arquivo) != 0) {
16 printf("Aviso: falha ao fechar o arquivo.\n");
17 return;
18 }
19 printf("Despesa registrada.\n");
20}

22void listarDespesas() {
23 char data[11], categoria[30];
24 double valor, total = 0.0;
25 int quantidade = 0;
26 FILE *arquivo = fopen("despesas.txt", "r");
27 if (arquivo == NULL) {
28 printf("Nenhum diario disponivel para leitura.\n");
29 return;
30 }
31 printf("\nDATA CATEGORIA VALOR\n");
32 while (fscanf(arquivo, "%10s %29s %lf", data, categoria, &valor) == 3) {
33 printf("%-11s %-25s R$ %8.2f\n", data, categoria, valor);
34 total += valor;
35 quantidade++;
36 }
37 if (ferror(arquivo)) printf("Erro durante a leitura.\n");
38 fclose(arquivo);
39 if (quantidade == 0) printf("Nenhuma despesa registrada.\n");
40 printf("Total: R$ %.2f\n", total);
41}
