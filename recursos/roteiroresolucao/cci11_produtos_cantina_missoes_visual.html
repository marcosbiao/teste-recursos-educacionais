Problema real

A cantina precisa controlar produtos com características diferentes

Cada produto possui nome, código, preço, quantidade e categoria. Durante o dia, chegam cinco sanduíches naturais e são vendidas três garrafas de água. O sistema deve atualizar somente a quantidade do produto correto e preservar os demais campos. Estoque inicial: água 18, sanduíche 6, suco 10 e salada de frutas 12 unidades.

1#include <stdio.h>

4typedef struct {
5 int codigo;
6 char nome[50];
7 double preco;
8 int quantidade;
9 char categoria[30];
10} Produto;
11
12int buscarProduto(Produto produtos[], int total, int codigo) {
13 for (int i = 0; i < total; i++) {
14 if (produtos[i].codigo == codigo) {
15 return i;
16 }
17 }
18 return -1;
19}
20
21int reporEstoque(Produto produtos[], int total, int codigo, int quantidade) {
22 int indice = buscarProduto(produtos, total, codigo);
23 if (indice == -1 || quantidade <= 0) {
24 return 0;
25 }
26 produtos[indice].quantidade += quantidade;
27 return 1;
28}
29
30int retirarEstoque(Produto produtos[], int total, int codigo, int quantidade) {
31 int indice = buscarProduto(produtos, total, codigo);
32 if (indice == -1 || quantidade <= 0 ||
33 produtos[indice].quantidade < quantidade) {
34 return 0;
35 }
36 produtos[indice].quantidade -= quantidade;
37 return 1;
38}
39
40void mostrarRelatorio(Produto produtos[], int total) {
41 for (int i = 0; i < total; i++) {
42 printf("%d | %s | R$ %.2f | %d | %s\n",
43 produtos[i].codigo, produtos[i].nome,
44 produtos[i].preco, produtos[i].quantidade,
45 produtos[i].categoria);
46 }
47}
